def get_boxscore_json(boxscore_url):
    r = requests.get(boxscore_url, headers=headers)
    soup = BeautifulSoup(r.text, 'html.parser')

    iframe = soup.select('wmt-stats-iframe')
    iframe_src = iframe[0].get('path')
    game_id = re.search(r'\d+', iframe_src)[0]

    api_link = (
        f'https://api.wmt.games/api/statistics/games/{game_id}'
        '?with[0]=actions&with[1]=players&with[2]=plays'
    )
    j = requests.get(api_link, headers=headers).json()
    html = r.text

    time_match = re.search(r'\b\d{1,2}:\d{2}\s?[ap]m\b', html, re.IGNORECASE)
    date_match = re.search(r'(Mon|Tue|Wed|Thu|Fri|Sat|Sun),?\s+[A-Z][a-z]{2}\.?\s+\d{1,2}\s+\(\d{4}\)', html)

    j["page_time"] = time_match.group(0) if time_match else None
    j["page_date"] = date_match.group(0) if date_match else None
    j["source_url"] = boxscore_url
    return j

# treat 11:00 a.m. as post-midnight for return time scoring since it likely indicates a late-night return that disrupts sleep more than an early-morning return
def normalize_return_time(t):
    if t is None:
        return None
    if isinstance(t, datetime.time):
        h, m = t.hour, t.minute
    else:
        parsed = pd.to_datetime(t, format="%I:%M:%S %p", errors="coerce")
        if pd.isna(parsed):
            return None
        h, m = parsed.hour, parsed.minute
    h = h + (24 if h < 11 else 0)
    return h * 60 + m

def density_score_for_value(key, value):
    if value is None:
        return None
    if key == "return_time":
        value = normalize_return_time(value)
        if value is None:
            return None
    for low, high, score in density_table[key]:
        if low <= value < high:
            return score
    return None

def compute_density_score(df_sb_travel):
    df_sb_travel["travel_density_score"] = (
        df_sb_travel["return_time"].apply(lambda x: density_score_for_value("return_time", x))
        + df_sb_travel["turnaround_time"].apply(lambda x: density_score_for_value("turnaround_time", x))
    )
    return df_sb_travel

def compute_residual_decay(df):
    if "turnaround_time" not in df.columns:
        raise KeyError("Missing required column: turnaround_time")

    df["half_life_days"] = df["trip_score"].apply(half_life_from_trip_score)
    df["residual"] = df["trip_score"] * np.exp(
        -np.log(2) * df["turnaround_time"] / df["half_life_days"]
    )
    return df
def compute_rtl(df):
    df = df.copy()
    df["travel_date_home"] = pd.to_datetime(df["travel_date_home"])

    start = df["travel_date_home"].min()
    end   = df["travel_date_home"].max() + pd.Timedelta(days=30)
    dates = pd.date_range(start, end, freq="D")

    rtl_rows = []
    for day in dates:
        active = df[df["travel_date_home"] <= day].copy()
        if active.empty:
            rtl_rows.append({"game_date": day, "rtl": 0.0})
            continue

        active["delta_days"] = (day - active["travel_date_home"]).dt.days
        active["daily_residual"] = active["trip_score"] * np.exp(
            -np.log(2) * active["delta_days"] / active["half_life_days"]
        )
        rtl_rows.append({"game_date": day, "rtl": active["daily_residual"].sum()})

    return pd.DataFrame(rtl_rows)

def compute_crtl(rtl_df):
    rtl_df = rtl_df.copy()
    rtl_df["crtl"] = rtl_df["rtl"].cumsum()
    return rtl_df
